CREATE TABLE student_details (id INT AUTO_INCREMENT UNIQUE,prn VARCHAR(20) PRIMARY KEY NOT NULL,name VARCHAR(50) NOT NULL,email VARCHAR(50) NOT NULL,password VARCHAR(255) NOT NULL,branch VARCHAR(50) NOT NULL,tenth DECIMAL(5,2),twelfth DECIMAL(5,2),cgpa1 DECIMAL(5,2) DEFAULT NULL,cgpa2 DECIMAL(5,2) DEFAULT NULL,cgpa3 DECIMAL(5,2) DEFAULT NULL,cgpa4 DECIMAL(5,2) DEFAULT NULL,cgpa5 DECIMAL(5,2) DEFAULT NULL,cgpa6 DECIMAL(5,2) DEFAULT NULL,cgpa7 DECIMAL(5,2) DEFAULT NULL,cgpa8 DECIMAL(5,2) DEFAULT NULL,aggregate DECIMAL(5,2) DEFAULT NULL,nationality VARCHAR(50),disabled ENUM('Yes', 'No'),gender ENUM('Male', 'Female'),phone_no VARCHAR(20),dob DATE,address TEXT,skills TEXT,placed ENUM('Yes', 'No'),created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP);

CREATE TABLE company_details (id INT AUTO_INCREMENT UNIQUE,company_id VARCHAR(50) PRIMARY KEY NOT NULL,company_name VARCHAR(100) NOT NULL,email VARCHAR(100) NOT NULL,contact_no VARCHAR(20) NOT NULL,address VARCHAR(200) NOT NULL,hr_name VARCHAR(100) NOT NULL,cutoff DECIMAL(5,2) NOT NULL,branch_type VARCHAR(50) NOT NULL,required_skills VARCHAR(200) NOT NULL,created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP);

CREATE TABLE applications (id INT AUTO_INCREMENT PRIMARY KEY, student_prn VARCHAR(50), company_id VARCHAR(50), company_name VARCHAR(100), cutoff DOUBLE, branch_type VARCHAR(50), student_name VARCHAR(100), student_branch VARCHAR(50), FOREIGN KEY (student_prn) REFERENCES student_details(prn), FOREIGN KEY (company_id) REFERENCES company_details(company_id));

CREATE TABLE admin (
    name VARCHAR(20) DEFAULT NULL,
    email VARCHAR(30) DEFAULT NULL,
    mobile_number MEDIUMTEXT DEFAULT NULL,
    faculty VARCHAR(20) DEFAULT NULL,
    prn INT NOT NULL PRIMARY KEY,
    password VARCHAR(20) DEFAULT NULL,
    rpassword VARCHAR(20) DEFAULT NULL
);
